154 private void handle_assignment(Binary_Expression e)
155 {
156 //For static variables, only do anything first time through
157 if(e.is_static && e.is_static_visited)
158 return;
159 else if(e.is_static)
160 e.is_static_visited = true;
161
162 //First, evaluate RHS to get value to assign to LHS
163 e.right.visit_with(this);
164 Value rhs = cur_value;
165
166 //Now, evaluate LHS to get the thing we're assigning to
167 e.left.visit_with(this);
168 Value lhs = cur_value;
169
170 //It's illegal to assign to a literal (i.e. 3 = 5)
171 if(lhs.is_literal)
172 error_out("Attempted to assign to a literal.");
173
174 //For #=, just copy the reference over, unless RHS is literal.
175 if(e.operator==Cantrip.Lexer.REF_ASSN)
176 {
177 //Error out if LHS invalid
178 if(cur_map==null && cur_ary==null)
179 error_out("Attempted to assign to an r-value.");
180
181 if(cur_map!=null)
182 cur_map.put(cur_str,(cur_value = copy_if_literal(rhs)));
183 else //cur_ary!=null
184 cur_ary[cur_ary_idx] = (cur_value = copy_if_literal(rhs));
185
186 cur_str = null;
187 cur_map = null;
188 }
189 else
190 //For =, if left-hand side is an aggregate, first look for an overload
191 if(lhs.type==Type.AGGREGATE && lhs.ag_value.containsKey("assign"))
192 operator_overload_helper(lhs,rhs,"assign"); //that's it then
193 else //If no overload found or left-hand side isn't an aggregate, invoke clone().
194 {
195 if(cur_map==null && cur_ary==null)
196 error_out("Attempted to assign to an r-value.");
197
198 //Clone does a one-level-deep copy of the right value and makes it the left value
199 if(cur_map!=null)
200 cur_map.put(cur_str,new Value(rhs));
201 else
202 cur_ary[cur_ary_idx] = new Value(rhs);
203
204 cur_str = null;
205 cur_map = null;
206 cur_ary = null;
207 }

210 private void process_primitives_operator(Value left, Value right, int op)
211 {
212 //Create result value and deal with type promotion rules
213 Value result = new Value();
214 result.type = left.type;
215 if(left.type==Type.CHARACTER && right.type!=Type.CHARACTER || //type promotion
216 left.type==Type.INTEGER && right.type==Type.FLOAT)
217 result.type = right.type;
218
219 if(result.type==Type.FLOAT && left.type!=Type.FLOAT)
220 result.float_value = do_operator(left.int_value,right.float_value,op);
221 else if(result.type==Type.FLOAT && right.type!=Type.FLOAT)
222 result.float_value = do_operator(left.float_value,right.int_value,op);
223 else if(result.type==Type.FLOAT)
224 result.float_value = do_operator(left.float_value,right.float_value,op);
225 else
226 result.int_value = do_operator(left.int_value,right.int_value,op);

232 result.type = Type.INTEGER;
233 result.int_value = (int)(result.float_value);
234 }

239 public void visit(Binary_Expression e)
240 {
241 //Special-case assignment and ref-assignment operators
242 if(e.operator=='=' || e.operator==Cantrip.Lexer.REF_ASSN)
243 {
244 handle_assignment(e);
245 return;
246 }
247
248 //Visit left position
249 e.left.visit_with(this);
250 Value left_value = cur_value;
251
252 //Short-circuit for && and ||
253 if(e.operator==Cantrip.Lexer.AND && left_value.int_value==0)
254 return;
255 else if(e.operator==Cantrip.Lexer.OR && left_value.int_value!=0)
256 return;
257
258 //Visit right position
259 e.right.visit_with(this);
260 Value right_value = cur_value;
261
262 //If we're not a number or aggregate, error out
263 if(left_value.type!=Type.CHARACTER && left_value.type!=Type.INTEGER &&
264 left_value.type!=Type.FLOAT && left_value.type!=Type.AGGREGATE)
265 error_out("Invalid left-hand side for binary expression operator "+e.operator);
266
267 //Handle operator overloading
268 if(left_value.type==Type.AGGREGATE)
269 {
270 String op_name = "";
271 switch(e.operator)
272 {
273 case '+': op_name = "add"; break;
274 case '-': op_name = "subtract"; break;
275 case '*': op_name = "multiply"; break;
276 case '/': op_name = "divide"; break;
277 case '%': op_name = "modulo"; break;
278 case Cantrip.Lexer.EQUALS: op_name = "equals"; break;
279 case Cantrip.Lexer.NOT_EQUALS:
280 operator_overload_helper(left_value,right_value,"equals");
281 cur_value.int_value = cur_value.int_value==0 ? 1 : 0;
282 break;
283 case '<': op_name = "lt"; break;
284 case Cantrip.Lexer.LT_EQUALS:
285 operator_overload_helper(left_value,right_value,"lt");
286 if(cur_value.int_value==0)
287 operator_overload_helper(left_value,right_value,"equals");
288 break;
289 case Cantrip.Lexer.GT_EQUALS:
290 operator_overload_helper(left_value,right_value,"lt");

292 break;

295 cur_value.int_value = cur_value.int_value==0 ? 1 : 0;

303 }

306 }
